        complete: boolean
        active: boolean
        created: Date               // started
        advent: '<date>'
        group: Group
        quest: Quest
            isPublic: boolean
            subquests: List<Quest>
            items: List<QuestItem>
                id: string
                groupAction: GroupAction
                config: { advent: '<date>', times: 100, percent: true, done: true }

            questItemId: 1
